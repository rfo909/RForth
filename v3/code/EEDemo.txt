10 CONSTANT BufLen
HERE BufLen allot CONSTANT Buf
0x50 CONSTANT I2CAddr
128 CONSTANT Loc              (EEPROM data location - page size is 64)

: Prep (n -- ) (prepare to read N bytes by setting the length byte)
	Buf writeb
;

: Init (initialize buffer to zero length)
	0 => i
	BEGIN
		0 Buf i + writeb
		i 1+ => i
	i BufLen < AGAIN?
;

: Add (byte -- )
	(byte) Buf Buf readb + 1+ writeb
	Buf readb 1+ Buf writeb
;

: AddW (word -- )
	dup 8 >> 0xFF & Add
	0xFF & Add
;
    
(set data)
: Loc Loc AddW ;
: Data 0xCAFE AddW 0xBADE AddW 0xCADA AddW ;

: Send Buf I2CAddr NATIVE I2C.masterWrite ;
: Wait I2CAddr NATIVE I2C.masterWWait ;
: Recv Buf I2CAddr NATIVE I2C.masterRead ;

(write to EEPROM)
: Write 
  Init Loc Data Send
  (wait a bit to give EEPROM time to update)
  Wait
;

(read from EEPROM)
: Read
  Init Loc Send (no wait, because no data written, only address into a register)
  6 Prep Recv (6 bytes to receive)
;


: Show (show Buf including length byte)
  0 => i 
  BEGIN
    Buf i + readb print cr
    i 1+ => i
    i BufLen < AGAIN?
;


: Test Write Read Show ;
