const true 1 :bool ;
const false 0 :bool ;

# Utility

: Reg! ( :byte =clearMask :byte =valueMask :int =addr -- )
	addr sys:read clearMask ! &
		valueMask clearMask & | addr sys:write ;

: wait ( :ulong =ms )
	 sys:millis ms + =ms loop{ sys:millis ms > break  } ;
