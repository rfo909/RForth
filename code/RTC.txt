Model number DS3231

( 734 bytes compiled to byte code ) 

(device i2c addr)
0x68 CONSTANT I2CAddr

(registers)

0x02 CONSTANT RH	(hours)
0x01 CONSTANT RM	(minutes)
0x00 CONSTANT RS	(seconds)

0x11 CONSTANT RTH	(temp high)
0x12 CONSTANT RTL	(temp low)

0x0e CONSTANT RCtrl	(control register)
0x0f CONSTANT RStat	(status register)

HERE 4 allot CONSTANT Buf

(from https://github.com/NorthernWidget/DS3231/blob/master/DS3231.cpp)
(hms are stored as binary coded decimal "bcd", need to convert)

: decToBcd (val--val)
	=> val
	val 10 / 16 * 
		val 10 % +
  ;
  
: bcdToDec (val--val)
	=> val
	val 16 / 10 *
		val 16 % +
;


(Read single register)
: Reg@ (register-- val)
	=> reg
	1 Buf writeb (bytes to write = register to read from)
	reg Buf 1+ writeb
	Buf I2CAddr NATIVE I2C.masterWrite
	
	1 Buf writeb  (number of bytes to read) 
	Buf I2CAddr NATIVE I2C.masterRead
	
	Buf 1+ readb
;

(Write single register byte)
: Reg! (val register --)
	=> reg 
	=> val
	2 Buf writeb (2 values to send)
	reg Buf 1+ writeb  (register address)
	val Buf 2 + writeb (value)
	
	Buf I2CAddr NATIVE I2C.masterWrite
;

: Set (hour minute sec -- )
	=> sec
	=> min
	=> hour

	RH Reg@ b11000000 & => topBits
	hour decToBcd  
		topBits | RH Reg!

	min decToBcd RM Reg!
	
	sec decToBcd RS Reg!
;

: Init
	(oscillator config)
	b00011100 RCtrl Reg!
	0 RStat Reg!

	(set 24h clock - bit in hours register)
	b01000000 inv => mask
	RH Reg@ mask & 
		RH Reg!
;

: H ( -- hours )
	RH Reg@ b00111111 & bcdToDec ;

: M ( -- minutes )
	RM Reg@ bcdToDec ;
	
: S ( -- seconds )
	RS Reg@ bcdToDec ;

(temperature in celsius)
: T ( -- temperature ) 
	(RTL deals with fractions, resolution 0.25 - ignoring)
	RTH Reg@
 ;
 

(Display time and temperature)

: Show
	cr 
	H print# ': .str
	M print# ': .str
	S print# "_ .str
	T print#s 'C .str
	
;
