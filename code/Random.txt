101 VARIABLE s     (small value)

(Overflow to initial value if too big)
: OFlow ( val limit -- val )
	=> limit
	dup limit > IF drop limit THEN

(28 bytes)
: Get (var init limit -- value)
	=> limit
	=> init 
	=> var 
	var @ 1+  (val)
	limit OFlow  (val)
	dup var !
;

: Big 
	b 31001 43033 Get 
;

: Small
	s 101 723 Get
;
