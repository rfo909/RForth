DictClear
Dict /Curses
/Curses DictUse

HERE 8 allot CONSTANT Buf

(initialize esc sequence)

: Esc 
	2 Buf writeb
	0x1b Buf 1+ writeb
	0x5b Buf 2 + writeb 
;	

: r+w (incr addr -- )
	dup => addr
	readb + addr writeb
;


: Str ( str -- ) 
	=> s
	s readb => len
	
	s 1+				(source)
		Buf dup readb + 1+	(target)
			len memcpy
			
	len Buf r+w
;

: Int ( n -- ) 10 Buf NATIVE Util.Itoa ;

: Out Buf .str ;
	

: cls Esc '2J Str Out ;
: home Esc 'H Str Out ;

: down ( n -- )
	Esc Int 'B Str Out ;
	
: right ( n -- )
	Esc Int 'C Str Out ;

: goto (y x --)
  home right down
;

DictClear
